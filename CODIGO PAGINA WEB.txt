# 2. Función de codificación de imagen
def encode_image(image_path):
    if image_path and os.path.exists(image_path):
        with open(image_path, "rb") as f:
            encoded = base64.b64encode(f.read()).decode('ascii')
            ext = str(image_path).split('.')[-1]
            mime_type = "jpeg" if ext.lower() == "jfif" else ext
        
            return f"data:image/{mime_type};base64,{encoded}"
    return ""

# --- AÑADIDO: LÓGICA DEL RANKING EUROPEO ---
ranking_equipos = []
try:
    if os.path.exists(ruta_excel_jugadores):
        df_jugadores = pd.read_excel(ruta_excel_jugadores)
        # Buscar las columnas de eventos exitosos o ganados
        cols_exitosos = [col for col in df_jugadores.columns if 'Successful' in col or 'Won' in col or 'won' in col.lower()]
        if cols_exitosos:
            # Agrupar por equipo y sumar
            df_equipos = df_jugadores.groupby('equipo')[cols_exitosos].sum().sum(axis=1).reset_index()
            df_equipos.columns = ['equipo', 'total_exitosos']
            df_equipos = df_equipos.sort_values('total_exitosos', ascending=False).reset_index(drop=True)
            df_equipos['ranking'] = df_equipos.index + 1
            ranking_equipos = df_equipos.to_dict('records')
except Exception as e:
    print(f"Error cargando el ranking: {e}")

def generar_lista_ranking(ranking_list, equipo_seleccionado=None):
    if not ranking_list:
        return [html.P("No se pudo cargar el ranking. Verifica el archivo Excel.", style={'color': 'red', 'textAlign': 'center'})]
    
    elementos = []
    for row in ranking_list:
        is_selected = (row['equipo'] == equipo_seleccionado)
        bg_color = 'rgba(0, 255, 204, 0.3)' if is_selected else 'transparent'
        border_color = '#00FFCC' if is_selected else '#333'
        
        elemento = html.Div(
            id=f"item-ranking-{str(row['equipo']).replace(' ', '_')}",
            style={
                'display': 'flex', 'justifyContent': 'space-between', 'padding': '15px', 
                'borderBottom': f'1px solid {border_color}', 'backgroundColor': bg_color,
                'fontWeight': 'bold' if is_selected else 'normal',
                'transition': '0.3s'
            },
            children=[
                html.Span(f"TOP {row['ranking']} - {row['equipo']}", style={'color': 'white' if not is_selected else '#00FFCC', 'fontSize': '18px'}),
                html.Span(f"{int(row['total_exitosos'])} eventos", style={'color': '#888' if not is_selected else '#00FFCC'})
            ]
        )
        elementos.append(elemento)
    return elementos

        elif prefijo == "atletico":
            ruta_archivo = ruta_atletico / nombre_archivo
        else:
            return dash.no_update
    else:
        ruta_archivo = ruta_betis / index_str
        
    if ruta_archivo.exists(): return dcc.send_file(str(ruta_archivo))
    return dash.no_update
